# 저자 별 카테고리 별 매출액 집계하기
```roomsql
SELECT 
    B.AUTHOR_ID, A.AUTHOR_NAME, B.CATEGORY, SUM(B.PRICE * BS.SALES) AS TOTAL_SALES
FROM 
    BOOK_SALES AS BS
    LEFT JOIN BOOK AS B ON B.BOOK_ID = BS.BOOK_ID
    LEFT JOIN AUTHOR AS A ON A.AUTHOR_ID = B.AUTHOR_ID
WHERE MONTH(BS.SALES_DATE) = "1" AND YEAR(BS.SALES_DATE) = '2022'
GROUP BY
    B.AUTHOR_ID, B.CATEGORY
ORDER BY
    B.AUTHOR_ID, B.CATEGORY DESC
```
